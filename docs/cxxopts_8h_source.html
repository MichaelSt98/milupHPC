572 void
573 checked_negate(R& r, T&& t, const std::string&, std::true_type)
574 {
575 // if we got to here, then `t` is a positive number that fits into
576 // `R`. So to avoid MSVC C4146, we first cast it to `R`.

588 template <typename T>
589 void
590 integer_parser(const std::string& text, T& value)
591 {
592 std::smatch match;
593 std::regex_match(text, match, integer_pattern);
594
595 if (match.length() == 0)
596 {
597 throw_or_mimic<argument_incorrect_type>(text);
598 }
599
600 if (match.length(4) > 0)
601 {
602 value = 0;
603 return;
604 }
605
606 using US = typename std::make_unsigned<T>::type;
607
608 constexpr bool is_signed = std::numeric_limits<T>::is_signed;
609 const bool negative = match.length(1) > 0;
610 const uint8_t base = match.length(2) > 0 ? 16 : 10;
611
612 auto value_match = match[3];
613
614 US result = 0;
615
616 for (auto iter = value_match.first; iter != value_match.second; ++iter)
617 {
618 US digit = 0;
619
620 if (*iter >= '0' && *iter <= '9')
621 {
622 digit = static_cast<US>(*iter - '0');
623 }
624 else if (base == 16 && *iter >= 'a' && *iter <= 'f')
625 {
626 digit = static_cast<US>(*iter - 'a' + 10);
627 }
628 else if (base == 16 && *iter >= 'A' && *iter <= 'F')
629 {
630 digit = static_cast<US>(*iter - 'A' + 10);
631 }
632 else
633 {
634 throw_or_mimic<argument_incorrect_type>(text);
635 }
636
637 const US next = static_cast<US>(result * base + digit);
638 if (result > next)
639 {
640 throw_or_mimic<argument_incorrect_type>(text);
641 }
642
643 result = next;
644 }
645
646 detail::check_signed_range<T>(negative, result, text);
647
648 if (negative)
649 {
650 checked_negate<T>(value, result, text, std::integral_constant<bool, is_signed>());
651 }
652 else
653 {
654 value = static_cast<T>(result);
655 }
656 }

1628 String
1629 format_description
1630 (
1631 const HelpOptionDetails& o,
1632 size_t start,
1633 size_t allowed,
1634 bool tab_expansion
1635 )
1636 {
1637 auto desc = o.desc;
1638
1639 if (o.has_default && (!o.is_boolean || o.default_value != "false"))
1640 {
1641 if(!o.default_value.empty())
1642 {
1643 desc += toLocalString(" (default: " + o.default_value + ")");
1644 }
1645 else
1646 {
1647 desc += toLocalString(" (default: \"\")");
1648 }
1649 }
1650
1651 String result;
1652
1653 if (tab_expansion)
1654 {
1655 String desc2;
1656 auto size = size_t{ 0 };
1657 for (auto c = std::begin(desc); c != std::end(desc); ++c)
1658 {
1659 if (*c == '\n')
1660 {
1661 desc2 += *c;
1662 size = 0;
1663 }
1664 else if (*c == '\t')
1665 {
1666 auto skip = 8 - size % 8;
1667 stringAppend(desc2, skip, ' ');
1668 size += skip;
1669 }
1670 else
1671 {
1672 desc2 += *c;
1673 ++size;
1674 }
1675 }
1676 desc = desc2;
1677 }
1678
1679 desc += " ";
1680
1681 auto current = std::begin(desc);
1682 auto previous = current;
1683 auto startLine = current;
1684 auto lastSpace = current;
1685
1686 auto size = size_t{};
1687
1688 bool appendNewLine;
1689 bool onlyWhiteSpace = true;
1690
1691 while (current != std::end(desc))
1692 {
1693 appendNewLine = false;
1694
1695 if (std::isblank(*previous))
1696 {
1697 lastSpace = current;
1698 }
1699
1700 if (!std::isblank(*current))
1701 {
1702 onlyWhiteSpace = false;
1703 }
1704
1705 while (*current == '\n')
1706 {
1707 previous = current;
1708 ++current;
1709 appendNewLine = true;
1710 }
1711
1712 if (!appendNewLine && size >= allowed)
1713 {
1714 if (lastSpace != startLine)
1715 {
1716 current = lastSpace;
1717 previous = current;
1718 }
1719 appendNewLine = true;
1720 }
1721
1722 if (appendNewLine)
1723 {
1724 stringAppend(result, startLine, current);
1725 startLine = current;
1726 lastSpace = current;
1727
1728 if (*previous != '\n')
1729 {
1730 stringAppend(result, "\n");
1731 }
1732
1733 stringAppend(result, start, ' ');
1734
1735 if (*previous != '\n')
1736 {
1737 stringAppend(result, lastSpace, current);
1738 }
1739
1740 onlyWhiteSpace = true;
1741 size = 0;
1742 }
1743
1744 previous = current;
1745 ++current;
1746 ++size;
1747 }
1748
1749 //append whatever is left but ignore whitespace
1750 if (!onlyWhiteSpace)
1751 {
1752 stringAppend(result, startLine, previous);
1753 }
1754
1755 return result;
1756 }
